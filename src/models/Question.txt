class QuestionSchema {
    /**
     * 
     * @param {String} id
     * @param {String} category 
     * @param {String} tq 
     * @param {String[]} choices 
     * @param {String} answer 
     */
    constructor( id, category, tq, choices, answer){
    this.id = id;
    this.category = category;
    this.tq = tq;
    this.choices = choices;
    this.answer = answer;
    }
    }
